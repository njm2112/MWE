\makeatletter

\def\parseline#1 "#2"{\@namedef{#2}{#1}}
\newcommand{\parseinput}[1]{%
  \def\@tempb{\par}%
  \openin\myinput=#1
  \loop\unless\ifeof\myinput
    \read\myinput to \@tempa
    \ifx\@tempa\@tempb\else
      \expandafter\parseline\@tempa
    \fi
  \repeat
  \closein\myinput
}
