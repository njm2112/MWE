\def\fibonacci#1{...
    \numbertimes=2\numone=0

    \loop
       \advance\numone  by  \numtwo  
       \savenumone=\the\numone  \numone=\numtwo  \numtwo=\savenumone 
       \advance\numbertimes  by 1 \ifnum \numbertimes<#1
    \repeat  
    \ifnum\numbertimes=#1 \advance\numone  by   \numtwo\fi 
    }
