% literate programming a short example
% from the lstlistings manual page 48
\documentclass{article}
\usepackage{graphicx}
\usepackage{listings,xcolor}
\begin{document}
\lstset{literate={:=}{{$\gets$}}1 
                      {<=}{{$\leq$}}1 
                      {>=}{{$\geq$}}1 
                      {<>}{{$\neq$}}1}
\begin{lstlisting}
var i:integer;
if (i<=0) i := 1;
if (i>=0) i := 0;
if (i<>0) i := 0;
if imagewidth>linewidth 
  let \a := \b;
  scaleimage
else
  do nothing;
fi
\end{lstlisting}
\end{document}
